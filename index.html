Hubungkan ke Google Sheets

Web app ini menggunakan Google Apps Script sebagai jembatan antara aplikasi dan Google Sheets Anda. Ikuti langkah di bawah sekali saja — setelah itu data scan akan otomatis tersinkronisasi.

1
Buka Google Sheets baru
Buka sheets.google.com → buat spreadsheet baru → beri nama misalnya "SMART-Boarding Log". Buat 3 sheet tab: ScanLog, Manifes, TransitLog.
2
Buat Google Apps Script
Di Google Sheets: klik menu Extensions → Apps Script. Hapus semua kode yang ada, paste kode berikut:
// ── SMART-Boarding Apps Script ── // Tempel kode ini ke Extensions → Apps Script, lalu Deploy sebagai Web App // Execute as: Me | Who has access: Anyone const SS = SpreadsheetApp.getActiveSpreadsheet(); // Fungsi pembantu: buat response dengan header CORS function makeResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // doGet: tes koneksi & handle preflight OPTIONS function doGet(e) { const action = e.parameter.action || ''; // Tes koneksi if (!action || action === 'ping') { return makeResponse({ ok: true, msg: 'SMART-Boarding API aktif', ts: new Date().toISOString() }); } // Ambil data manifes (untuk load dari perangkat petugas) if (action === 'getManifes') { try { const sh = SS.getSheetByName('Manifes'); if (!sh) { return makeResponse({ ok: false, msg: 'Sheet Manifes tidak ditemukan — Admin belum upload manifes', rows: [] }); } const lastRow = sh.getLastRow(); const lastCol = sh.getLastColumn(); if (lastRow < 2 || lastCol < 1) { return makeResponse({ ok: false, msg: 'Sheet Manifes kosong ('+lastRow+' baris, '+lastCol+' kolom)', rows: [] }); } const rows = sh.getDataRange().getValues(); // Validasi baris pertama adalah header (bukan data penumpang) // Header biasanya tidak mengandung angka 15 digit (nomor tiket) const firstRow = rows[0] ? rows[0].map(String) : []; const looksLikeData = firstRow.some(cell => /^\d{15}$/.test(cell.trim())); if (looksLikeData) { return makeResponse({ ok: false, msg: 'Sheet Manifes tidak memiliki baris header · Upload ulang manifes dari Admin', rows: [], debug: { firstRow: firstRow, issue: 'missing_header' } }); } return makeResponse({ ok: true, rows: rows, debug: { totalRows: rows.length, headers: firstRow, sheet: 'Manifes' } }); } catch(err) { return makeResponse({ ok: false, msg: 'Error getManifes: ' + err.message, rows: [] }); } } // Ambil scan log untuk dashboard Admin if (action === 'getScanLog') { try { const sh = SS.getSheetByName('ScanLog'); if (!sh || sh.getLastRow() < 2) { return makeResponse({ ok: true, rows: [], naik: 0, f1: 0, dup: 0, total: 0 }); } const rows = sh.getDataRange().getValues(); const headers = rows[0].map(String); const data = rows.slice(1).map(row => { const o = {}; headers.forEach((h, i) => { o[h.trim()] = String(row[i]||'').trim(); }); return o; }); const naik = data.filter(r => r['Status'] === 'NAIK').length; const f1 = data.filter(r => r['Pos'] === 'F1' && r['Status'] === 'OK').length; const dup = data.filter(r => r['Status'] === 'DUPLIKAT').length; return makeResponse({ ok: true, rows: data, naik, f1, dup, total: data.length }); } catch(err) { return makeResponse({ ok: false, msg: 'Error: ' + err.message }); } } return makeResponse({ ok: false, msg: 'Action tidak dikenal' }); } // doPost: terima data dari web app function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === 'addLog') { const sh = getOrCreateSheet('ScanLog', ['Waktu','No Tiket','Nama','Pos','Status','Info','Tipe Kapal','Nama Kapal','Voyage']); sh.appendRow([ data.time, data.ticket, data.name, data.pos, data.status, data.info || '', data.shipType || '', data.shipName || '', data.voyage || '' ]); } if (data.action === 'addTransit') { const sh = getOrCreateSheet('TransitLog', ['No Tiket','Nama','Waktu Keluar','Waktu Kembali','Durasi','Status','Timestamp']); if (data.status === 'kembali') { // Cari baris keluar yang sudah ada, update status & waktu kembali const vals = sh.getDataRange().getValues(); let updated = false; for (let i = vals.length - 1; i >= 1; i--) { if (String(vals[i][0]).trim() === String(data.ticket).trim() && vals[i][5] === 'keluar') { let durasi = ''; try { const [oh,om,os] = String(vals[i][2]).split(':').map(Number); const [rh,rm,rs] = String(data.returnTime).split(':').map(Number); const diffSec = (rh*3600+rm*60+rs) - (oh*3600+om*60+os); if (diffSec >= 0) { const m = Math.floor(diffSec/60); const s = diffSec % 60; durasi = m > 0 ? m+'m '+s+'d' : s+'d'; } } catch(err) {} sh.getRange(i+1, 4).setValue(data.returnTime || ''); sh.getRange(i+1, 5).setValue(durasi); sh.getRange(i+1, 6).setValue('kembali'); sh.getRange(i+1, 7).setValue(new Date().toLocaleTimeString()); updated = true; break; } } if (!updated) { sh.appendRow([data.ticket, data.name, data.outTime||'', data.returnTime||'', '', 'kembali', new Date().toLocaleTimeString()]); } } else { // Keluar: tambah baris baru sh.appendRow([data.ticket, data.name, data.outTime||'', '', '', 'keluar', new Date().toLocaleTimeString()]); } } if (data.action === 'addLanjutan') { const sh = getOrCreateSheet('PenumpangLanjutan', ['No Tiket','Waktu','Tanggal','Kapal','No Voyage','Timestamp']); sh.appendRow([ data.ticket, data.waktu || '', data.tanggal || '', data.kapal || '', data.voyage || '', new Date().toLocaleTimeString() ]); } if (data.action === 'uploadManifes') { // ── ARSIP SCANLOG LAMA sebelum sesi baru dimulai ── const scanSh = SS.getSheetByName('ScanLog'); if (scanSh && scanSh.getLastRow() > 1) { try { // Buat nama sheet arsip dengan timestamp const now = new Date(); const stamp = Utilities.formatDate(now, 'Asia/Makassar', 'yyyyMMdd_HHmm'); const arsipName = 'ScanLog_' + stamp; // Copy sheet ScanLog ke arsip const arsipSh = scanSh.copyTo(SS); arsipSh.setName(arsipName); // Warnai tab arsip abu-abu agar mudah dibedakan arsipSh.setTabColor('#9E9E9E'); // Kosongkan ScanLog aktif - hapus semua lalu tulis ulang header scanSh.clearContents(); const scanHeaders = ['Waktu','No Tiket','Nama','Pos','Status','Info','Tipe Kapal','Nama Kapal','Voyage']; scanSh.appendRow(scanHeaders); scanSh.getRange(1, 1, 1, scanHeaders.length) .setBackground('#0A2342') .setFontColor('#FFFFFF') .setFontWeight('bold'); // Catat arsip di sheet log internal (opsional) Logger.log('ScanLog diarsip ke: ' + arsipName); } catch(archErr) { Logger.log('Gagal arsip ScanLog: ' + archErr.message); // Lanjut upload manifes meski arsip gagal } } // ── TULIS MANIFES BARU ── const sh = getOrCreateSheet('Manifes', data.headers); // Hapus SEMUA isi sheet (termasuk header lama) sh.clearContents(); // Tulis header di baris 1 sh.appendRow(data.headers); sh.getRange(1, 1, 1, data.headers.length) .setBackground('#0A2342') .setFontColor('#FFFFFF') .setFontWeight('bold'); // Tulis data penumpang mulai baris 2 data.rows.forEach(r => sh.appendRow(r)); } if (data.action === 'updateStatus') { // Update status penumpang di sheet Manifes const sh = SS.getSheetByName('Manifes'); if (sh) { const vals = sh.getDataRange().getValues(); const headerRow = vals[0]; const ticketCol = headerRow.findIndex(h => String(h).toLowerCase().includes('tiket')); const statusCol = headerRow.findIndex(h => String(h).toLowerCase() === 'status'); if (ticketCol >= 0 && statusCol >= 0) { for (let i = 1; i < vals.length; i++) { if (String(vals[i][ticketCol]).trim() === String(data.ticket).trim()) { sh.getRange(i + 1, statusCol + 1).setValue(data.status); break; } } } } } return makeResponse({ ok: true }); } catch (err) { return makeResponse({ ok: false, error: err.message }); } } // Buat sheet baru dengan header jika belum ada function getOrCreateSheet(name, headers) { let sh = SS.getSheetByName(name); if (!sh) { // Sheet belum ada - buat baru sh = SS.insertSheet(name); } // Selalu pastikan baris 1 berisi header yang benar if (headers && headers.length > 0) { const firstRow = sh.getLastRow() > 0 ? sh.getRange(1, 1, 1, headers.length).getValues()[0] : []; const hasHeader = firstRow.some(cell => String(cell).trim() !== ''); if (!hasHeader) { // Baris 1 kosong - tulis header if (sh.getLastRow() === 0) { sh.appendRow(headers); } else { sh.getRange(1, 1, 1, headers.length).setValues([headers]); } sh.getRange(1, 1, 1, headers.length) .setBackground('#0A2342') .setFontColor('#FFFFFF') .setFontWeight('bold'); } } return sh; }
3
Deploy sebagai Web App
Di Apps Script: klik Deploy → New deployment → pilih type Web app.
⚠️ Wajib diatur persis seperti ini:
• Execute as → Me (email Anda)
• Who has access → Anyone (bukan "Anyone with Google account")

Klik Deploy → klik Authorize access → pilih akun Google Anda → klik Allow.
Copy URL yang muncul (format: https://script.google.com/macros/s/AKfy.../exec)

⚠️ Setiap ada perubahan kode, Deploy ulang via Deploy → Manage deployments → Edit (ikon pensil) → Version: New version → Deploy.

🔧
Troubleshooting Koneksi
❌ "Failed to fetch" atau error koneksi?
Penyebab paling umum adalah pengaturan deploy yang salah di Apps Script.
Gunakan jika ada data lama yang mengganggu atau setelah update versi aplikasi.
Checklist wajib:
☐  Who has access diset ke Anyone (bukan "Anyone with Google account")
☐  Sudah klik Authorize access dan klik Allow saat pertama deploy
☐  URL yang dipakai adalah URL dari hasil Deploy, bukan URL editor Apps Script
☐  Setelah ubah kode, sudah deploy ulang dengan New version
☐  Koneksi internet HP/laptop aktif
Cara tes manual URL Apps Script:
Buka URL Apps Script di browser baru. Jika muncul teks JSON seperti:
{"ok":true,"msg":"SMART-Boarding API aktif"}
berarti Apps Script sudah berjalan benar dan URL bisa dipakai.
Kenapa pakai content-type text/plain?
Google Apps Script tidak mendukung CORS preflight (OPTIONS request) yang dikirim browser saat content-type = application/json. Solusinya adalah mengirim data dengan header text/plain — Apps Script tetap bisa membaca isi JSON-nya, dan browser tidak mengirim preflight request.
